Punto 6

Se elaboró un código similar al COMPUTER EXAMPLE C6.2 que se muestra a continuación

t0=-0.5;
tf=1.5;
t=linspace (t0, tf,1000);
x = @(t) 6.*t.*(t>=-0.5 & t<0.5)+6.*(1-t).*(t>=0.5 & t<1.5);
t = linspace (-2*pi, 2*pi,1000);
y=x(t);
sumterms = zeros(16, length(t));
gei Ilinspace();
sumterms(1,:) = 1/2;
for n = 1:size(sumterms,1)-1;
sumterms(n+1,:) = 24*sin(n*pi/2)*sin(pi*n*t)/(n^2*pi^2);
end
x_N = cumsum (sumterms);
figure(1);
clf;
ind = 0;
for N = [0,1:2:size(sumterms, 1)-1]
ind = ind+1;
subplot (3,3,ind);
plot (t,x_N(N+1,:),t,y,'k--');
axis ([-1 2 -4 4]);
xlabel('t');
aux=strcat('x_{',num2str(N),'}');
ylabel(aux);
end
