  figure
  x=[0 -9 -6 -3 0 3 6 9 0 0];
  h=[0 4 2 0 2 4 0 0 0 0];
  n=-4:5;
  subplot(3,1,1);
  stem(n,x);
  xlabel('n');
  ylabel('x[n]');
  subplot(3,1,2);
  stem(n,h);
  xlabel('n');
  ylabel('h[n]');
  n=(0:1:length(x)+length(h)-2);
  subplot(3,1,3);
  c=conv(x,h);
  stem(n,c,'k');
  grid on
  title('conv(x,h)')
  xlabel('n');
  ylabel('y[n]');
